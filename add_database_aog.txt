
EXEC sp_rename 'Employees.Address', 'HomeAddress', 'COLUMN';

-- Step 1: Preview all columns that contain '#' (review before running)
SELECT 
    TABLE_SCHEMA,
    TABLE_NAME,
    COLUMN_NAME,
    REPLACE(COLUMN_NAME, '#', '') AS NEW_COLUMN_NAME
FROM 
    INFORMATION_SCHEMA.COLUMNS
WHERE 
    COLUMN_NAME LIKE '%#%'
ORDER BY 
    TABLE_SCHEMA, TABLE_NAME, COLUMN_NAME;
